    61тЖТ    61тЖТ// тФАтФА Client-side CSV export тФАтФА
    62тЖТ    62тЖТfunction exportToCSV(rows: ContactRow[], filename: string) {
    63тЖТ    63тЖТ  const esc = (v: string | number | null | undefined) => {
    64тЖТ    64тЖТ    if (v == null) return '""'
    65тЖТ    65тЖТ    const s = String(v)
    66тЖТ    66тЖТ    return s.includes(',') || s.includes('"') || s.includes('\n') ? `"${s.replace(/"/g, '""')}"` : s
    67тЖТ    67тЖТ  }
    68тЖТ    68тЖТ  const header = 'Name,Email,Job Title,Company,Health,Status\n'
    69тЖТ    69тЖТ  const body = rows.map(r =>
    70тЖТ    70тЖТ    [esc(r.name), esc(r.email), esc(r.jobTitle), esc(r.company?.name), esc(r.emailHealth), esc(r.status)].join(',')
    71тЖТ    71тЖТ  ).join('\n')
    72тЖТ    72тЖТ  const blob = new Blob(['\uFEFF' + header + body], { type: 'text/csv' })
    73тЖТ    73тЖТ  const url = URL.createObjectURL(blob)
    74тЖТ    74тЖТ  const a = document.createElement('a')
    75тЖТ    75тЖТ  a.href = url
    76тЖТ    76тЖТ  a.download = filename
    77тЖТ    77тЖТ  a.click()
    78тЖТ    78тЖТ  URL.revokeObjectURL(url)
    79тЖТ    79тЖТ}
