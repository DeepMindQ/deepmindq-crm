   119→function makeContacts() {
   120→  const contacts: Array<{
   121→    id: string; companyId: string; name: string; email: string;
   122→    jobTitle: string; roleBucket: string; linkedinUrl?: string;
   123→    phone?: string; location?: string; status: string;
   124→    emailHealth: string; emailHealthScore: number;
   125→    lastContactedAt?: Date; lastValidatedAt?: Date; archivedAt?: Date;
   126→  }> = [];
   127→
   128→  // Role bucket distribution: decision_maker 20%, influencer 30%, end_user 30%, blocker 10%, champion 10%
   129→  const roleDist: string[] = [
   130→    'decision_maker','influencer','influencer','end_user','end_user','end_user','blocker','champion',
   131→  ];
   132→  // Email health distribution: valid 60%, risky 15%, invalid 10%, unknown 15%
   133→  const healthDist: string[] = [
   134→    'valid','valid','valid','valid','valid','valid','risky','risky','invalid','unknown','unknown',
   135→  ];
   136→
   137→  const namePool: Array<{ first: string; last: string }> = [];
   138→  const used = new Set<string>();
   139→  for (const f of FIRST_NAMES) {
   140→    for (const l of LAST_NAMES) {
   141→      const key = `${f} ${l}`;
   142→      if (!used.has(key)) { used.add(key); namePool.push({ first: f, last: l }); }
   143→    }
   144→  }
   145→  // Shuffle deterministically
   146→  let seed = 42;
   147→  function rng() { seed = (seed * 16807 + 0) % 2147483647; return (seed - 1) / 2147483646; }
   148→  for (let i = namePool.length - 1; i > 0; i--) {
   149→    const j = Math.floor(rng() * (i + 1));
   150→    [namePool[i], namePool[j]] = [namePool[j], namePool[i]];
   151→  }
   152→
   153→  let nameIdx = 0;
   154→  let contactNum = 1;
   155→
   156→  for (const comp of companies) {
   157→    const count = 4 + Math.floor(rng() * 2); // 4-5
   158→    const domain = comp.domain || 'example.com';
   159→
   160→    for (let i = 0; i < count; i++) {
   161→      const person = namePool[nameIdx++ % namePool.length];
   162→      const emailPattern = rng();
   163→      let email: string;
   164→      if (emailPattern < 0.33) {
   165→        email = `${person.first.toLowerCase()}@${domain}`;
   166→      } else if (emailPattern < 0.66) {
   167→        email = `${person.first.toLowerCase()}.${person.last.toLowerCase()}@${domain}`;
   168→      } else {
   169→        email = `${person.first[0].toLowerCase()}${person.last.toLowerCase()}@${domain}`;
   170→      }
   171→
   172→      const title = JOB_TITLES[Math.floor(rng() * JOB_TITLES.length)];
   173→      const role = roleDist[Math.floor(rng() * roleDist.length)];
   174→      const health = healthDist[Math.floor(rng() * healthDist.length)];
   175→      const healthScore = health === 'valid' ? 85 + Math.floor(rng() * 15)
   176→        : health === 'risky' ? 45 + Math.floor(rng() * 25)
   177→        : health === 'invalid' ? Math.floor(rng() * 15)
   178→        : 0;
   179→
   180→      const isArchived = rng() < 0.05; // 5% archived
   181→
   182→      contacts.push({
   183→        id: `cont-${contactNum++}`,
   184→        companyId: comp.id,
   185→        name: `${person.first} ${person.last}`,
   186→        email,
   187→        jobTitle: title,
   188→        roleBucket: role,
   189→        linkedinUrl: `https://linkedin.com/in/${person.first.toLowerCase()}-${person.last.toLowerCase()}`,
   190→        phone: `+1 ${Math.floor(rng()*900+100)} ${Math.floor(rng()*900+100)} ${Math.floor(rng()*9000+1000)}`,
   191→        location: comp.location,
   192→        status: 'new',
   193→        emailHealth: health,
   194→        emailHealthScore: healthScore,
   195→        lastContactedAt: rng() < 0.4 ? daysAgo(Math.floor(rng() * 30) + 1) : undefined,
   196→        lastValidatedAt: rng() < 0.6 ? daysAgo(Math.floor(rng() * 14) + 1) : undefined,
   197→        archivedAt: isArchived ? daysAgo(Math.floor(rng() * 60) + 30) : undefined,
   198→      });
   199→    }
   200→  }
   201→
   202→  return contacts;
   203→}
