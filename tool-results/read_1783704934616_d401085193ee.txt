    64→
    65→function matchScoreClasses(score: number | null | undefined) {
    66→  if (score == null) return 'bg-gray-100 text-gray-600 border border-gray-200'
    67→  if (score >= 80) return 'bg-emerald-50 text-emerald-700 border border-emerald-200'
    68→  if (score >= 60) return 'bg-amber-50 text-amber-700 border border-amber-200'
    69→  return 'bg-red-50 text-red-700 border border-red-200'
    70→}
    71→
    72→function confidenceClasses(confidence: string | undefined) {
    73→  switch (confidence) {
    74→    case 'high': return 'bg-emerald-50 text-emerald-700 border border-emerald-200'
    75→    case 'medium': return 'bg-amber-50 text-amber-700 border border-amber-200'
    76→    case 'low': return 'bg-red-50 text-red-700 border border-red-200'
    77→    default: return 'bg-gray-100 text-gray-600 border border-gray-200'
    78→  }
    79→}
    80→
    81→function emailHealthBadge(health: string) {
    82→  switch (health) {
    83→    case 'valid': return { label: 'Valid', cls: 'bg-emerald-50 text-emerald-700 border-emerald-200' }
    84→    case 'risky': return { label: 'Risky', cls: 'bg-amber-50 text-amber-700 border-amber-200' }
    85→    case 'invalid': return { label: 'Invalid', cls: 'bg-red-50 text-red-700 border-red-200' }
    86→    default: return { label: 'Unknown', cls: 'bg-gray-100 text-gray-600 border-gray-200' }
    87→  }
    88→}
    89→
    90→function providerDisplayName(provider: string) {
    91→  switch (provider) {
    92→    case 'openai': return 'OpenAI'
    93→    case 'gemini': return 'Google Gemini'
    94→    case 'groq': return 'Groq'
    95→    default: return provider
    96→  }
    97→}

   146→/* ═══════════════════════════════════════════════════════════════
   147→   Toggle Group — reusable for Tone / Length / CTA

   149→
   150→function ToggleGroup<T extends string>({
   151→  label,
   152→  description,
   153→  options,
   154→  value,
   155→  onChange,
   156→}: {
   157→  label: string
   158→  description: string
   159→  options: { value: T; label: string }[]
   160→  value: T
   161→  onChange: (v: T) => void
   162→}) {
   163→  return (
   164→    <div className="bg-white rounded-xl border border-gray-200/80 card-rest p-4 md:p-6 space-y-3">
   165→      <div>
   166→        <h3 className="text-sm font-semibold text-gray-900">{label}</h3>
   167→        <p className="text-xs text-gray-500 mt-0.5">{description}</p>
   168→      </div>
   169→      <div className="flex items-center gap-2 flex-wrap">
   170→        {options.map(opt => (
   171→          <button
   172→            key={opt.value}
   173→            onClick={() => onChange(opt.value)}
   174→            className={cn(
   175→              'px-4 py-2 rounded-lg text-sm font-medium transition-all duration-150',
   176→              value === opt.value ? TOGGLE_ACTIVE : TOGGLE_INACTIVE,
   177→            )}
   178→          >
   179→            {opt.label}
   180→          </button>
   181→        ))}
   182→      </div>
   183→    </div>
   184→  )
   185→}
