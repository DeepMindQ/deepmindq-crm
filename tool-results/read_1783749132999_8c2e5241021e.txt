    64→
    65→function matchScoreClasses(score: number | null | undefined) {
    66→  if (score == null) return 'bg-gray-100 text-gray-600 border border-gray-200'
    67→  if (score >= 80) return 'bg-emerald-50 text-emerald-700 border border-emerald-200'
    68→  if (score >= 60) return 'bg-amber-50 text-amber-700 border border-amber-200'
    69→  return 'bg-red-50 text-red-700 border border-red-200'
    70→}
    71→
    72→function confidenceClasses(confidence: string | undefined) {
    73→  switch (confidence) {
    74→    case 'high': return 'bg-emerald-50 text-emerald-700 border border-emerald-200'
    75→    case 'medium': return 'bg-amber-50 text-amber-700 border border-amber-200'
    76→    case 'low': return 'bg-red-50 text-red-700 border border-red-200'
    77→    default: return 'bg-gray-100 text-gray-600 border border-gray-200'
    78→  }
    79→}
    80→
    81→function emailHealthBadge(health: string) {
    82→  switch (health) {
    83→    case 'valid': return { label: 'Valid', cls: 'bg-emerald-50 text-emerald-700 border-emerald-200' }
    84→    case 'risky': return { label: 'Risky', cls: 'bg-amber-50 text-amber-700 border-amber-200' }
    85→    case 'invalid': return { label: 'Invalid', cls: 'bg-red-50 text-red-700 border-red-200' }
    86→    default: return { label: 'Unknown', cls: 'bg-gray-100 text-gray-600 border-gray-200' }
    87→  }
    88→}
    89→
    90→function providerDisplayName(provider: string) {
    91→  switch (provider) {
    92→    case 'openai': return 'OpenAI'
    93→    case 'gemini': return 'Google Gemini'
    94→    case 'groq': return 'Groq'
    95→    default: return provider
    96→  }
    97→}

   151→/* ═══════════════════════════════════════════════════════════════
   152→   Toggle Group — reusable for Tone / Length / CTA
   153→   ═══════════════════════════════════════════════════════════════ */
   154→
   155→function ToggleGroup<T extends string>({
   156→  label,
   157→  description,
   158→  options,
   159→  value,
   160→  onChange,
   161→}: {
   162→  label: string
   163→  description: string
   164→  options: { value: T; label: string }[]
   165→  value: T
   166→  onChange: (v: T) => void
   167→}) {
   168→  return (
   169→    <div className="bg-white rounded-xl border border-gray-200/80 card-rest p-4 md:p-6 space-y-3">
   170→      <div>
   171→        <h3 className="text-sm font-semibold text-gray-900">{label}</h3>
   172→        <p className="text-xs text-gray-500 mt-0.5">{description}</p>
   173→      </div>
   174→      <div className="flex items-center gap-2 flex-wrap">
   175→        {options.map(opt => (
   176→          <button
   177→            key={opt.value}
   178→            onClick={() => onChange(opt.value)}
   179→            className={cn(
   180→              'px-4 py-2 rounded-lg text-sm font-medium transition-all duration-150',
   181→              value === opt.value ? TOGGLE_ACTIVE : TOGGLE_INACTIVE,
   182→            )}
   183→          >
   184→            {opt.label}
   185→          </button>
   186→        ))}
   187→      </div>
   188→    </div>
   189→  )
   190→}

   196→export default function EmailGenerationScreen() {

   198→

   200→  const [localContactId, setLocalContactId] = useState<string | null>(selectedContactId)
   201→  const contactId = localContactId
   202→
   203→  // Reason: we intentionally read `selectedContactId` only on mount to capture a navigated-in

   205→  // highlighted. Re-running this effect when selectedContactId changes would defeat the purpose.
   206→  useEffect(() => {
   207→    if (selectedContactId) {
   208→      setSelectedContactId(null)
   209→    }
   210→  }, [])
   211→
   212→  // ── Search debounce ──
   213→  const timerRef = useRef<ReturnType<typeof setTimeout>>(undefined)
   214→  const [contactSearch, setContactSearch] = useState('')
   215→  const [debouncedSearch, setDebouncedSearch] = useState('')
   216→
   217→  useEffect(() => {
   218→    return () => {
   219→      if (timerRef.current) clearTimeout(timerRef.current)
   220→    }
   221→  }, [])
   222→
   223→  const handleSearchChange = (val: string) => {
   224→    setContactSearch(val)
   225→    if (timerRef.current) clearTimeout(timerRef.current)
   226→    timerRef.current = setTimeout(() => setDebouncedSearch(val), 300)
   227→  }
   228→
   229→  // ── Preferences for defaults ──
   230→  const { data: prefs, error: prefsError } = useQuery({
   231→    queryKey: ['preferences'],
   232→    queryFn: () => fetch('/api/preferences').then(r => {
   233→      if (!r.ok) throw new Error('Failed to load preferences')
   234→      return r.json()
   235→    }),
   236→  })
   237→
   238→  // ── Sync defaults from prefs once loaded ──
   239→  const [tone, setTone] = useState<Tone>('professional-casual')
   240→  const [emailLength, setEmailLength] = useState<EmailLength>('medium')
   241→  const [ctaStyle, setCtaStyle] = useState<CtaStyle>('soft')
   242→  const [generatedSubject, setGeneratedSubject] = useState('')
   243→  const [generatedBody, setGeneratedBody] = useState('')
   244→  const [lastDraftId, setLastDraftId] = useState<string | null>(null)
   245→  const [lastMatchScore, setLastMatchScore] = useState<number | null>(null)
   246→  const [lastConfidence, setLastConfidence] = useState<string | undefined>(undefined)
   247→  const [draftContactId, setDraftContactId] = useState<string | null>(null)
   248→  const [isAiGenerated, setIsAiGenerated] = useState(false)
   249→  const [aiProviderName, setAiProviderName] = useState('')
   250→  const [subjectSaving, setSubjectSaving] = useState(false)
   251→
   252→  useEffect(() => {
   253→    if (prefs) {
   254→      setTone((prefs.tone as Tone) || 'professional-casual')
   255→      setEmailLength((prefs.emailLength as EmailLength) || 'medium')
   256→      setCtaStyle((prefs.ctaStyle as CtaStyle) || 'soft')
   257→    }
   258→  }, [prefs])
   259→
   260→  // ── Contacts ──
   261→  const { data: contactsData, isLoading: contactsLoading, error: contactsError } = useQuery({
   262→    queryKey: ['contacts', 'email-gen-sidebar', debouncedSearch],
   263→    queryFn: () => {
   264→      const p = new URLSearchParams({ pageSize: '50' })
   265→      if (debouncedSearch) p.set('search', debouncedSearch)
   266→      return fetch(`/api/contacts?${p}`).then(r => {
   267→        if (!r.ok) throw new Error('Failed to load contacts')
   268→        return r.json()
   269→      })
   270→    },
   271→  })
   272→
   273→  const contacts = (contactsData?.contacts ?? []) as ContactRow[]
   274→
   275→  // ── Selected contact (may need separate fetch if navigated from elsewhere) ──
   276→  const selectedContact = contacts.find(c => c.id === contactId) ?? null
   277→
   278→  const { data: preselectedContact, isLoading: preselectedLoading, error: preselectedError } = useQuery({
   279→    queryKey: ['contact', contactId],
   280→    queryFn: () => fetch(`/api/contacts/${contactId}`).then(r => {
   281→      if (!r.ok) throw new Error('Failed to load contact')
   282→      return r.json()
   283→    }),
   284→    enabled: !!contactId && !selectedContact,
   285→  })
   286→
   287→  const activeContact = selectedContact || (preselectedContact ? {
   288→    id: preselectedContact.id,
   289→    name: preselectedContact.name,
   290→    email: preselectedContact.email,
   291→    jobTitle: preselectedContact.jobTitle,
   292→    company: preselectedContact.company ? { id: preselectedContact.company.id, name: preselectedContact.company.name } : null,
   293→    emailHealth: preselectedContact.emailHealth || 'unknown',
   294→    emailHealthScore: preselectedContact.emailHealthScore,
   295→    companyId: preselectedContact.companyId,
   296→  } : null)
   297→
   298→  const hasDraft = generatedBody.length > 0 && draftContactId === contactId
   299→
   300→  // ── Navigation helpers ──
   301→  const goToSettings = () => setActiveView('settings')
   302→  const goToContactProfile = () => {
   303→    if (activeContact) {
   304→      setSelectedContactId(activeContact.id)
   305→      setActiveView('contact-profile')
   306→    }
   307→  }
   308→  const goToCompanyProfile = (companyId: string) => {
   309→    setSelectedCompanyId(companyId)
   310→    setActiveView('company-profile')
   311→  }
   312→
   313→  const clearDraft = () => {
   314→    setGeneratedSubject('')
   315→    setGeneratedBody('')
   316→    setLastDraftId(null)
   317→    setLastMatchScore(null)
   318→    setLastConfidence(undefined)
   319→    setDraftContactId(null)
   320→    setIsAiGenerated(false)
   321→    setAiProviderName('')
   322→  }
   323→
   324→  // ── Generate mutation ──
   325→  const generateMutation = useMutation({
   326→    mutationFn: () =>
   327→      fetch(`/api/contacts/${contactId}/generate-email`, {
   328→        method: 'POST',
   329→        headers: { 'Content-Type': 'application/json' },
   330→        body: JSON.stringify({ tone, emailLength, ctaStyle }),
   331→      })
   332→        .then(r => r.ok ? r.json() : r.json().then((e: { error?: string }) => { throw new Error(e.error || 'Generation failed') })),
   333→    onSuccess: (data: { subject?: string; body?: string; draftId?: string; matchScore?: number; confidence?: string }) => {
   334→      setGeneratedSubject(data.subject || '')
   335→      setGeneratedBody(data.body || '')
   336→      setLastDraftId(data.draftId || null)
   337→      setLastMatchScore(data.matchScore ?? null)
   338→      setLastConfidence(data.confidence)
   339→      setDraftContactId(contactId || null)
   340→      // Determine if AI was used based on prefs
   341→      const aiKey = prefs?.aiApiKey
   342→      setIsAiGenerated(!!aiKey)
   343→      setAiProviderName(providerDisplayName(prefs?.aiProvider || 'openai'))
   344→      toast.success('Email generated successfully')
   345→    },
   346→    onError: (e: Error) => toast.error(e.message),
   347→  })
   348→
   349→  // ── Save edited subject ──
   350→  const saveSubjectMutation = useMutation({
   351→    mutationFn: ({ draftId, subject }: { draftId: string; subject: string }) =>
   352→      fetch(`/api/drafts/${draftId}`, {
   353→        method: 'PATCH',
   354→        headers: { 'Content-Type': 'application/json' },
   355→        body: JSON.stringify({ subject }),
   356→      })
   357→        .then(r => r.ok ? r.json() : r.json().then((e: { error?: string }) => { throw new Error(e.error || 'Failed to save subject') })),
   358→    onSuccess: () => {
   359→      toast.success('Subject updated')
   360→      setSubjectSaving(false)
   361→    },
   362→    onError: (e: Error) => {
   363→      toast.error(e.message)
   364→      setSubjectSaving(false)
   365→    },
   366→  })

   373→

   391→  }

   393→  return (
   394→    <div className="flex h-full -m-6 flex-col md:flex-row">

   396→          Mobile Contact Selector
   397→         ═══════════════════════════════════════════════════════ */}

   399→        <Select value={contactId || ''} onValueChange={(id) => {

   404→            <SelectValue placeholder="Select a contact..." />
   405→          </SelectTrigger>
   406→          <SelectContent>
   407→            {contacts.map((c) => (

   414→        </Select>
   415→        <AiStatusBanner onGoToSettings={goToSettings} />

   417→
   418→      {/* ═══════════════════════════════════════════════════════
   419→          Left Sidebar (hidden on mobile)
   420→         ═══════════════════════════════════════════════════════ */}
   421→      <aside className="hidden md:flex w-80 border-r border-gray-200/80 bg-gray-50/50 flex-col shrink-0">
   422→        {/* AI Status Banner */}
   423→        <div className="p-4 border-b border-gray-200/60 bg-white">
   424→          <AiStatusBanner onGoToSettings={goToSettings} />

   427→        {/* Selected Contact Card */}

   430→            <div className="flex items-start gap-3">

   439→                  {activeContact.name}
   440→                </button>
   441→                <p className="text-xs text-gray-500 font-mono truncate mt-0.5">{activeContact.email}</p>
   442→                {activeContact.company && (
   443→                  <button

   459→            </div>

   482→              {[...Array(5)].map((_, i) => (
   483→                <div key={i} className="h-16 rounded-lg bg-gray-100 animate-pulse" />

   485→            </div>
   486→          ) : contactsError ? (
   487→            <div className="px-4 py-6 text-center">
   488→              <AlertTriangle className="size-6 text-red-400 mx-auto mb-2" />
   489→              <p className="text-xs text-red-500">Failed to load contacts</p>
   490→            </div>
   491→          ) : contacts.length === 0 ? (
   492→            <div className="px-4 py-8 text-center">
