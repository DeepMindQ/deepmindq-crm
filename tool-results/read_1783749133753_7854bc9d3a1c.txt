    55→// ── Client-side CSV export ──
    56→function exportToCSV(rows: CompanyRow[], filename: string) {
    57→  const esc = (v: string | number | null | undefined) => {
    58→    if (v == null) return '""'
    59→    const s = String(v)
    60→    return s.includes(',') || s.includes('"') || s.includes('\n') ? `"${s.replace(/"/g, '""')}"` : s
    61→  }
    62→  const header = 'Name,Domain,Industry,Score,Status,Employee Size,Freshness\n'
    63→  const body = rows.map(r =>
    64→    [esc(r.name), esc(r.domain), esc(r.industry), r.intelligenceScore ?? 0, esc(r.status), esc(r.employeeSize), esc(r.dataFreshness)].join(',')
    65→  ).join('\n')
    66→  const blob = new Blob(['\uFEFF' + header + body], { type: 'text/csv' })
    67→  const url = URL.createObjectURL(blob)
    68→  const a = document.createElement('a')
    69→  a.href = url
    70→  a.download = filename
    71→  a.click()
    72→  URL.revokeObjectURL(url)
    73→}
