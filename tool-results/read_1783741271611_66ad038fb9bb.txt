    24→function hoursAgo(n: number): Date {
    25→  return new Date(Date.now() - n * HOUR)

    28→function randomFrom<T>(arr: T[]): T {
    29→  return arr[Math.floor(Math.random() * arr.length)]
    30→}
    31→
    32→function randomInt(min: number, max: number): number {
    33→  return Math.floor(Math.random() * (max - min + 1)) + min
