    94→function SettingsToggleGroup<T extends string>({
    95→  label,
    96→  description,
    97→  options,
    98→  value,
    99→  onChange,
   100→}: {
   101→  label: string
   102→  description?: string
   103→  options: { value: T; label: string }[]
   104→  value: T
   105→  onChange: (v: T) => void
   106→}) {
   107→  return (
   108→    <div>
   109→      <label className="text-xs font-medium text-gray-700 block mb-1.5">{label}</label>
   110→      {description && <p className="text-[11px] text-gray-400 mb-2">{description}</p>}
   111→      <div className="flex items-center gap-2 flex-wrap">
   112→        {options.map(opt => (
   113→          <button
   114→            key={opt.value}
   115→            onClick={() => onChange(opt.value)}
   116→            className={cn(
   117→              'px-3.5 py-1.5 rounded-lg text-xs font-medium transition-all duration-150',
   118→              value === opt.value ? TOGGLE_ACTIVE : TOGGLE_INACTIVE,
   119→            )}
   120→          >
   121→            {opt.label}
   122→          </button>
   123→        ))}
   124→      </div>
   125→    </div>
   126→  )
   127→}
