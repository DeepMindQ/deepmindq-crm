   111→function getInitials(name?: string | null): string {
   112→  if (!name) return '?'
   113→  const parts = name.trim().split(/\s+/)
   114→  if (parts.length >= 2) return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase()
   115→  return parts[0][0].toUpperCase()
   116→}
