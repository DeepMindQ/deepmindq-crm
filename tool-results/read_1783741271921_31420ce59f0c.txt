   114→function pick<T>(arr: T[]): T {
   115→  return arr[Math.floor(Math.random() * arr.length)];
   116→}
   117→
   118→function pickWeighted<T extends string>(arr: T[], weights: number[]): T {
   119→  const total = weights.reduce((a, b) => a + b, 0);
   120→  let r = Math.random() * total;
   121→  for (let i = 0; i < arr.length; i++) {
   122→    r -= weights[i];
   123→    if (r <= 0) return arr[i];
   124→  }
   125→  return arr[arr.length - 1];
   126→}
