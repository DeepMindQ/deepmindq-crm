    88→
    89→function SettingsToggleGroup<T extends string>({
    90→  label,
    91→  description,
    92→  options,
    93→  value,
    94→  onChange,
    95→}: {
    96→  label: string
    97→  description?: string
    98→  options: { value: T; label: string }[]
    99→  value: T
   100→  onChange: (v: T) => void
   101→}) {
   102→  return (
   103→    <div>
   104→      <label className="text-xs font-medium text-gray-700 block mb-1.5">{label}</label>
   105→      {description && <p className="text-[11px] text-gray-400 mb-2">{description}</p>}
   106→      <div className="flex items-center gap-2 flex-wrap">
   107→        {options.map(opt => (
   108→          <button
   109→            key={opt.value}
   110→            onClick={() => onChange(opt.value)}
   111→            className={cn(
   112→              'px-3.5 py-1.5 rounded-lg text-xs font-medium transition-all duration-150',
   113→              value === opt.value ? TOGGLE_ACTIVE : TOGGLE_INACTIVE,
   114→            )}
   115→          >
   116→            {opt.label}
   117→          </button>
   118→        ))}
   119→      </div>
   120→    </div>
   121→  )
   122→}
