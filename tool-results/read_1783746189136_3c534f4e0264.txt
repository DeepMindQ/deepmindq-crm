    93→function SettingsToggleGroup<T extends string>({
    94→  label,
    95→  description,
    96→  options,
    97→  value,
    98→  onChange,
    99→}: {
   100→  label: string
   101→  description?: string
   102→  options: { value: T; label: string }[]
   103→  value: T
   104→  onChange: (v: T) => void
   105→}) {
   106→  return (
   107→    <div>
   108→      <label className="text-xs font-medium text-gray-700 block mb-1.5">{label}</label>
   109→      {description && <p className="text-[11px] text-gray-400 mb-2">{description}</p>}
   110→      <div className="flex items-center gap-2 flex-wrap">
   111→        {options.map(opt => (
   112→          <button
   113→            key={opt.value}
   114→            onClick={() => onChange(opt.value)}
   115→            className={cn(
   116→              'px-3.5 py-1.5 rounded-lg text-xs font-medium transition-all duration-150',
   117→              value === opt.value ? TOGGLE_ACTIVE : TOGGLE_INACTIVE,
   118→            )}
   119→          >
   120→            {opt.label}
   121→          </button>
   122→        ))}
   123→      </div>
   124→    </div>
   125→  )
   126→}
